import java.util.*;
public class Main{
    public static void main(String args[])
    {
    	Scanner scan=new Scanner(System.in);
        int n=scan.nextInt();
        int array[]=new int[n];
        for(int i=0;i<n;i++) {
        	array[i]=scan.nextInt();
        }
        Arrays.parallelSort(array);
        for(int x:array) {
        	System.out.print(x+" ");
        }
    }
}
