一个简单的结构体代码，认知结构体地址传参与结构体传参的方式
#include<stdio.h>
#include<windows.h>
struct student
{
	char name[20];
	int age;
	char sex[5];
	char id[20];
};
struct student s1 = { "张三", 12, "男", "11111111" };
struct student s2 = { "李四", 12, "男", "22222222" };

void print1(struct student  s1)
{
	printf("%s\n", s1.name);

}
void print2(struct student* ps)
{
	printf("%s\n", ps->name);

}

int main()
{
	print1(s1);
	print2(&s2);
system("pause");
	return 0;
	
}
